    /**
     * Does the given sequence of characters form a valid email address? 
     * The test is done using RFC 822's regex.
     * @param value the sequence of characters to test.
     * @return {@code true} for valid, {@code false} for invalid.
     */
    public static boolean isValidEmail(String value) {
        Pattern regex = Pattern.compile(ValidationRegexPattern);
        return regex.matcher(value).matches();
    }
    
    protected Email(String wrappedValue) {
        super(wrappedValue);
    }

    @Override
    public Either<String, String> validate(String value) {
        return isValidEmail(value) ? right(value) 
                                   : left("invalid email: " + value);
    }
